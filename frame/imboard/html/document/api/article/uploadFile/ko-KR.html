#{lan.example}
$(file).on("change", function(e)
{
	var formData = new FormData();
	var files = e.target.files;
	var length = files.length;
	for(var i=0; i<length; i++)
	{
		formData.append("file-" + new Date().getTime() + "-" + i, files[i]);
	}
	
	var param =
	{
		data : 	attachImageData,
		success : function(result)
		{
			console.error("파일업로드 성공 : ", result);
		},
		error : function(result)
		{
			console.error("파일업로드 에러 : ", result);
		}
	};
	
	$.api.article.uploadFile(param);
});
#{lan.parameter}
변수명 변수타입 설명
data 필수 String 첨부할 파일 데이터
success 필수 Integer 업로드 성공 콜백함수
error 필수 String 업로드 실패 콜백함수
#{lan.data}
변수명 변수타입 설명
code Integer 통신결과코드
message Integer 통신결과메시지
data String[] 업로드한 파일의 URL
