#{lan.example}
$(file).on("change", function(e)
{
	var formData = new FormData();
	var files = e.target.files;
	var length = files.length;
	for(var i=0; i<length; i++)
	{
		formData.append("file-" + new Date().getTime() + "-" + i, files[i]);
	}
	
	var param =
	{
		data : 	attachImageData,
		success : function(result)
		{
			console.error("Success : ", result);
		},
		error : function(result)
		{
			console.error("Error : ", result);
		}
	};
	
	$.api.article.uploadFile(param);
});
#{lan.parameter}
Variable's name Variable's type Description
data Required String File data
success Required Integer Success callback
error Required String Error callback
#{lan.data}
Variable's name Variable's type Description
code Integer Response code
message Integer Response message
data String[] URL of uploaded files.
