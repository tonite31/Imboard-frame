3-3-1. $.api.board.getBoardList

#{lan.example}
var result = $.api.board.getBoardList();
if(result.code == 1000)
	console.log(result.data);
else
	console.error(result);
#{lan.data}
변수명 변수타입 설명
code Integer 통신결과코드
message Integer 통신결과메시지
data boardList id String 게시판 아이디
name String 게시판 이름
priority Integer 우선순위
description String 설명
useYn String 사용여부
viewListLevel Integer 게시판 글 목록 보기 권한레벨
viewDetailLevel Integer 게시글 보기 권한레벨
writeLevel Integer 게시글 작성 권한레벨
writeCommentLevel Integer 덧글 작성 권한레벨
creator String 게시판 생성 사용자 아이디
