3-7-4. $.api.user.insertUser

#{lan.example}
$.api.user.insertUser({id : "test", name : "테스트용 사용자 등록", writerProfileImgUrl : "http://localhost:3000/test.jpg"});
#{lan.parameter}
Variable's name Variable's type Description
id Required String id
displayId String nickname
name String name
writerProfileImgUrl String Profile image url
level Integer Level
userStatus String (Y or N) Account status.
Y : Available
N : Disabled
#{lan.data}
Variable's name Variable's type Description
code Integer Response code
message Integer Response message
