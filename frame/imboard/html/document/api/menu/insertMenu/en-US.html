$.api.menu.insertMenu

#{lan.example}
$.api.menu.insertMenu({id : "test", name : "Menu for test.", type : 0, parameter : "?id=test", priority : 0, viewLevel : 10});
#{lan.parameter}
Variable's name Variable's type Description
id Required String id
name String name
type Integer type
parameter String Url querystring. use this, if you connect menu to board. ex)?boardId=test
priority Integer Priority
parentMenuId String Id of parent menu.
viewLevel Integer User's level of viewing a menu.
childMenuCount Integer count of child menus.
creator String User's id that was created a menu.
#{lan.data}
Variable's name Variable's type Description
code Integer Response code
message Integer Response message
